float GetFractionPart(float Number)
{
    return Number-(int)Number;
}

int MyCeilMethod1(float Number)
{
    int IntPart=(int)Number;

    if (abs(GetFractionPart(Number))>0)
    {
        if(Number>0)
            return IntPart+1;
        else
            return IntPart; 
    }else
        return IntPart;
    
}
