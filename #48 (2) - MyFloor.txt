#include<iostream>
using namespace std;

float MyFloor(float Number)
{
	if (Number > 0)
		return int(Number);
	else	   
		return int(--Number);
}

float ReadNumber()
{
	float Number;

	cout << "Please enter a number ?\n";
	cin >> Number;
	return Number;
}

int main()
{
	float Number = ReadNumber();

	cout << "\nMy MyFloor Result : " << MyFloor(Number);
	cout << "\nC++ floor Result : " << floor(Number);
	return 0;
}
