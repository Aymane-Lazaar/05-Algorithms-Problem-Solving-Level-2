#include<iostream>
using namespace std;

float MyABS(float Number)
{
	if (Number < 0.0)
		return (Number * -1.0);
	else
		return Number;
}

float ReadNumber()
{
	float Number;

	cout << "Please enter a number ?\n";
	cin >> Number;
	return Number;
}

int main()
{
	float Number = ReadNumber();

	cout << "\nMy abs Result : " << MyABS(Number) ;
	cout << "\nC++ abs Result : " << abs(Number) ;
	return 0;
}
