float GetFractionPart(float Number)
{
    return Number-(int)Number;
}

float Myabs(float number){
    if (number>0)
        return number;
    else
        return number * -1;
    
}

//method1
int MyRoundMethod1(float number)
{

    int IntPart;
    IntPart=(int)number;

    float FractionsPart=GetFractionPart(number);

    if (Myabs(FractionsPart)>=0.5)
    {
        if(number>0)
            return ++IntPart;
        else
            return --IntPart;
    }else
        return IntPart;
}
